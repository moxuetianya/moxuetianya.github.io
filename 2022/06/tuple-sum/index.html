#include <tuple>
#include <type_traits>
#include <iostream>
#include <vector>
#include <string>

using namespace std::string_literals;
template <typename T>
double filter(T arg)
{
  if constexpr (std::is_same_v<T, char>) {
    return 0.0;
  } else if constexpr (std::is_integral_v<T> || std::is_floating_point_v<T>) {
    return arg;
  } else {
    return 0;
  }
}

template <typename ...Ts>
double tuple_sum(const std::tuple<Ts...>& tpl) {
  return std::apply([](auto&&... args) {return (filter(args) + ...); }, tpl);
}
